/* ---------------- CRC16(Modbus) ---------------- */
static unsigned short CRC16_Modbus(const unsigned char* data, unsigned int length)
{
    unsigned short crc = 0xFFFF;
    for (unsigned int i = 0; i < length; i++) {
        crc ^= data[i];
        for (unsigned int j = 0; j < 8; j++) {
            if (crc & 0x0001)
                crc = (crc >> 1) ^ 0xA001;
            else
                crc >>= 1;
        }
    }
    return crc;
}

/* ---------------- 帧构造 ---------------- */
static std::string BuildWriteCmd(uint8_t id, uint16_t reg, int16_t value)
{
    unsigned char buf[8];
    buf[0] = id;
    buf[1] = 0x06;
    buf[2] = static_cast<unsigned char>(reg >> 8);
    buf[3] = static_cast<unsigned char>(reg & 0xFF);
    buf[4] = static_cast<unsigned char>((value >> 8) & 0xFF);
    buf[5] = static_cast<unsigned char>(value & 0xFF);
    unsigned short crc = CRC16_Modbus(buf, 6);
    buf[6] = static_cast<unsigned char>(crc & 0xFF);     // CRC low
    buf[7] = static_cast<unsigned char>(crc >> 8);       // CRC high
    return std::string(reinterpret_cast<char*>(buf), 8);
}

static std::string BuildReadCmd(uint8_t id, uint16_t regAddr, uint16_t regCount)
{
    unsigned char buf[8];
    buf[0] = id;
    buf[1] = 0x03;
    buf[2] = static_cast<unsigned char>(regAddr >> 8);
    buf[3] = static_cast<unsigned char>(regAddr & 0xFF);
    buf[4] = static_cast<unsigned char>(regCount >> 8);
    buf[5] = static_cast<unsigned char>(regCount & 0xFF);
    unsigned short crc = CRC16_Modbus(buf, 6);
    buf[6] = static_cast<unsigned char>(crc & 0xFF);
    buf[7] = static_cast<unsigned char>(crc >> 8);
    return std::string(reinterpret_cast<char*>(buf), 8);
}

/* ---------------- 解析 + 校验 ---------------- */
static bool VerifyModbusCRC(const std::string& frame)
{
    if (frame.size() < 3) return false;
    unsigned short calc = CRC16_Modbus(
        reinterpret_cast<const unsigned char*>(frame.data()),
        static_cast<unsigned int>(frame.size() - 2)
    );
    const unsigned char* p = reinterpret_cast<const unsigned char*>(frame.data());
    unsigned char crcl = static_cast<unsigned char>(calc & 0xFF);
    unsigned char crch = static_cast<unsigned char>((calc >> 8) & 0xFF);
    return p[frame.size() - 2] == crcl && p[frame.size() - 1] == crch;
}

static uint16_t ParseReadResponseU16(const std::string& resp)
{
    // 响应: id 0x03 0x02 dataHi dataLo CRClo CRChi
    if (resp.size() != 7) return 0;
    const unsigned char* d = reinterpret_cast<const unsigned char*>(resp.data());
    return static_cast<uint16_t>( (static_cast<uint16_t>(d[3]) << 8) | d[4] );
}

/* ---------------- 基础 I/O：用 SerialPort::writeBytes + read ---------------- */
// 0x06 写单寄存器，读取 8 字节回显并校验
static bool ModbusWriteAndEcho(SerialPort& sp, const std::string& frame, DWORD timeout_ms = 300)
{
    const unsigned char* out = reinterpret_cast<const unsigned char*>(frame.data());
    if (!sp.writeBytes(out, frame.size())) {
        std::cerr << "[ERR] Write failed\n";
        return false;
    }
    std::string echo = sp.read(8, timeout_ms);
    if (echo.size() != 8) {
        std::cerr << "[ERR] Echo len=" << echo.size() << " (expect 8)\n";
        return false;
    }
    if (!VerifyModbusCRC(echo)) {
        std::cerr << "[ERR] Echo CRC error\n";
        return false;
    }
    // 回显前 6 字节应与请求前 6 字节一致
    if (echo.compare(0, 6, frame, 0, 6) != 0) {
        std::cerr << "[ERR] Echo mismatch\n";
        return false;
    }
    return true;
}

// 0x03 读 1 个寄存器（7 字节响应）
static bool ModbusReadU16(SerialPort& sp, uint8_t id, uint16_t addr, uint16_t& out, DWORD timeout_ms = 300)
{
    std::string req = BuildReadCmd(id, addr, 1);
    const unsigned char* outBuf = reinterpret_cast<const unsigned char*>(req.data());
    if (!sp.writeBytes(outBuf, req.size())) {
        std::cerr << "[ERR] Read request write failed\n";
        return false;
    }
    std::string resp = sp.read(7, timeout_ms);
    if (resp.size() != 7) {
        std::cerr << "[ERR] Read resp len=" << resp.size() << " (expect 7)\n";
        return false;
    }
    if (!VerifyModbusCRC(resp)) {
        std::cerr << "[ERR] Read resp CRC error\n";
        return false;
    }
    const unsigned char* d = reinterpret_cast<const unsigned char*>(resp.data());
    if (d[0] != id || d[1] != 0x03 || d[2] != 0x02) {
        std::cerr << "[ERR] Read resp header error\n";
        return false;
    }
    out = ParseReadResponseU16(resp);
    return true;
}
static bool ModbusWriteMaybeEcho(SerialPort& sp,
                                 const std::string& frame,
                                 bool wait_echo = true,
                                 DWORD timeout_ms = 300)
{
    const unsigned char* out = (const unsigned char*)frame.data();
    if (!sp.writeBytes(out, frame.size())) {
        std::cerr << "[ERR] Write failed\n";
        return false;
    }
    if (!wait_echo) return true;

    std::string echo = sp.read(16, timeout_ms);
    if (echo.size() < 8) {
        std::cerr << "[WARN] Echo len=" << echo.size() << " (<8)\n";
        return true; // 不把它当致命错误
    }

    std::string head8 = echo.substr(0, 8);
    if (!VerifyModbusCRC(head8)) {
        std::cerr << "[WARN] Echo CRC error\n";
        return true;
    }
    if (head8.compare(0, 6, frame, 0, 6) != 0) {
        std::cerr << "[WARN] Echo mismatch\n";
    }
    return true;
}



/* ---------------- 业务流程 ---------------- */
int main()
{
    SerialPort sp;
    if (!sp.open("COM4", 9600, NOPARITY, 8, TWOSTOPBITS)) {
        std::cerr << "Open COM4 failed\n";
        return 1;
    }

    const uint8_t id = 1;       // PA-71
    const int16_t speed  = 20;   // rpm (0x0204 只能正值)
    const int16_t turns  = 2;  // 0x0202 可正负（负数表示反向）
    const int16_t pulses = 0;   // 0x0203 可正负（圈内脉冲）


    // 坐标清零会正常回显
    // ModbusWriteMaybeEcho(sp, BuildWriteCmd(id, 0x0122, 1), /*wait_echo=*/true);
    // std::this_thread::sleep_for(std::chrono::milliseconds(200));


    // ② 柔化加减速（避免位置超差）
    ModbusWriteAndEcho(sp, BuildWriteCmd(id, 0x0028, 2000)); // 加速(ms)
    ModbusWriteAndEcho(sp, BuildWriteCmd(id, 0x0029, 2000)); // 减速(ms)

    // ③ 设置速度/位置
    ModbusWriteAndEcho(sp, BuildWriteCmd(id, 0x0204, speed));
    ModbusWriteAndEcho(sp, BuildWriteCmd(id, 0x0202, turns));
    ModbusWriteAndEcho(sp, BuildWriteCmd(id, 0x0203, pulses));

    // ④ 启动（沿触发：先 0 再 1）
    ModbusWriteAndEcho(sp, BuildWriteCmd(id, 0x011F, 0));
    std::this_thread::sleep_for(std::chrono::milliseconds(100));
    ModbusWriteAndEcho(sp, BuildWriteCmd(id, 0x011F, 1));

    std::cout << "Motor started: rotating " << turns << " turns\n";

    // ⑤ 读取报警/状态/位置偏差
    uint16_t alarm=0, status=0, errLo=0, errHi=0;
    ModbusReadU16(sp, id, 0x1013, alarm); // 报警码
    ModbusReadU16(sp, id, 0x1010, status);// 输出状态（bit2=定位完成, bit1=报警）
    ModbusReadU16(sp, id, 0x1005, errLo); // 位置偏差低 16 位
    ModbusReadU16(sp, id, 0x1006, errHi); // 位置偏差高 16 位
    uint32_t pos_err = (static_cast<uint32_t>(errHi) << 16) | errLo;

    std::cout << "Alarm=" << alarm
              << "  Status=0x" << std::hex << status << std::dec
              << "  PosErr=" << pos_err << "\n";

    // ⑥ 等待执行完成（演示）
    std::this_thread::sleep_for(std::chrono::seconds(5));
    std::cout << "Motion finished.\n";

    sp.close();
    return 0;
}
