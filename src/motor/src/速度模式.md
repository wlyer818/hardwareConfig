#include "serial.h"
#include <iostream>
#include <thread>
#include <iomanip>
#include <vector>

/* ---------------- CRC16(Modbus) ---------------- */
unsigned short CRC16_Modbus(const unsigned char* data, unsigned int length)
{
    unsigned short crc = 0xFFFF;
    for (unsigned int i = 0; i < length; i++) {
        crc ^= data[i];
        for (unsigned int j = 0; j < 8; j++) {
            if (crc & 0x0001)
                crc = (crc >> 1) ^ 0xA001;
            else
                crc >>= 1;
        }
    }
    return crc;
}

/* ---------------- 构造写单寄存器命令 (功能码 0x06) ---------------- */
std::vector<unsigned char> BuildWriteCmd(uint8_t id, uint16_t reg, uint16_t value)
{
    std::vector<unsigned char> buf(8);
    buf[0] = id;
    buf[1] = 0x06;
    buf[2] = reg >> 8;
    buf[3] = reg & 0xFF;
    buf[4] = value >> 8;
    buf[5] = value & 0xFF;

    unsigned short crc = CRC16_Modbus(buf.data(), 6);
    buf[6] = crc & 0xFF;
    buf[7] = crc >> 8;
    return buf;
}

/* ---------------- 发送命令函数 ---------------- */
bool SendCmd(SerialPort& port, const std::vector<unsigned char>& cmd)
{
    std::string hex;
    for (auto b : cmd)
        std::cout << std::hex << std::setw(2) << std::setfill('0') << int(b) << " ";
    std::cout << std::dec << std::endl;

    return port.writeBytes(cmd.data(), cmd.size());
}

/* ---------------- 主函数 ---------------- */
int main()
{
    SerialPort port;
    if (!port.open("COM4", CBR_9600, NOPARITY, 8, ONESTOPBIT)) {
        std::cerr << "串口打开失败！" << std::endl;
        return -1;
    }

    uint8_t id = 1;  // 设备地址

    // 1️⃣ 设置速度
    auto setSpeed = BuildWriteCmd(id, 0x0015, 100);  // 速度100
    SendCmd(port, setSpeed); 
    std::this_thread::sleep_for(std::chrono::milliseconds(100));

    // 2️⃣ 设置加速/减速时间
    auto accel = BuildWriteCmd(id, 0x0028, 50);  // 加速时间50ms
    SendCmd(port, accel);
    std::this_thread::sleep_for(std::chrono::milliseconds(100));

    auto decel = BuildWriteCmd(id, 0x0029, 50);  // 减速时间50ms
    SendCmd(port, decel);
    std::this_thread::sleep_for(std::chrono::milliseconds(100));

    // 3️⃣ 正转运行
    std::cout << "[INFO] 正转启动..." << std::endl;
    auto forward = BuildWriteCmd(id, 0x011F, 1);
    SendCmd(port, forward);
    std::this_thread::sleep_for(std::chrono::seconds(3));

    // 4️⃣ 反转运行
    std::cout << "[INFO] 反转启动..." << std::endl;
    auto reverse = BuildWriteCmd(id, 0x011F, 2);
    SendCmd(port, reverse);
    std::this_thread::sleep_for(std::chrono::seconds(3));

    // 5️⃣ 停止运行
    std::cout << "[INFO] 停止..." << std::endl;
    auto stop = BuildWriteCmd(id, 0x011F, 0);
    SendCmd(port, stop);

    port.close();
    std::cout << "[INFO] 完成控制流程。" << std::endl;
    return 0;
}
